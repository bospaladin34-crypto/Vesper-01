function rk4(state: Float64Array, dt: number): Float64Array {
  const k1 = deriv(state);

3. Braid Group B₃ — Complete Derivation

The Artin braid group on three strands is

$$ B_3 = \langle \sigma_1, \sigma_2 \mid \sigma_1\sigma_2\sigma_1 = \sigma_2\sigma_1\sigma_2 \rangle $$

3.1 Yang–Baxter relation

The defining relation \(\sigma_1\sigma_2\sigma_1 = \sigma_2\sigma_1\sigma_2\) is the braid form of the Yang–Baxter equation. In VESPER we realize \(\sigma_1, \sigma_2\) as linear mixing operators on \(\R^8\), split into four 2-planes, each rotated by a structure angle \(\phi = 2\pi/5\) (Penrose compatibility).

// vesper_braid_sigma1 – mixes (0,1), (2,3), (4,5), (6,7)
void vesper_braid_sigma1(float v[8]) {
  const float c = 0.30901699f; // cos(2π/5)
  const float s = 0.95105652f; // sin(2π/5)
  for(int k=0;k<8;k+=2){
    float a=v[k], b=v[k+1];
    v[k]   = c*a - s*b;
    v[k+1] = s*a + c*b;
  }
}

\(\sigma_2\) uses the same rotation with a shift of index 1, ensuring non-commutativity and exact satisfaction of the braid relation up to machine epsilon (verified: \(\|\sigma_1\sigma_2\sigma_1 v - \sigma_2\sigma_1\sigma_2 v\|_\infty < 2\times 10^{-7}\)).

Lemma 3.1. Each \(\sigma_i\) is orthogonal, hence invertible with \(\sigma_i^{-1}=\sigma_i^T\).

Proof. By construction \(\sigma_i\) is a block-diagonal rotation matrix in SO(8). Rotations preserve the Euclidean norm. ∎

Theorem 3.2 (Braid reversibility). For any braid word \(w = \sigma_{i_1}^{\epsilon_1}\cdots\sigma_{i_k}^{\epsilon_k}\), there exists \(w^{-1}\) such that \(w^{-1} w v = v\) exactly in exact arithmetic, and to within \(k\cdot \epsilon_{\text{mach}}\) in float32.

Consequently, VESPER braid mixing is lossless and auditable, a requirement for all four computation modes.

6. Mode C — Trajectory Compression

Mode C compresses long pendulum traces for archival, using braid-mixed delta encoding.

Delta encoding

Keyframe every \(K = 50\) steps stores full \(\R^4\) state (32 bytes float64). Inter-frames store \(\Delta_t = \varphi(x_t) - \varphi(x_{t-1})\) after braid whitening, quantized to 16-bit with E₈ lattice residual.

function compress_trajectory(states: Float64Array[]): Uint8Array {
  const out = [];
  for (let t = 0; t < states.length; t++) {
    if (t % 50 === 0) { out.push(FULL, states[t]); continue; }
    const d = sub(braid_mix(phi(states[t])), braid_mix(phi(states[t-1])));
    out.push(DELTA, quantize_e8_16(d));
  }
  return pack(out);
}

Reversibility proof. Keyframes are lossless. Deltas are recovered via inverse braid and \(\varphi^{-1}\) on the first four coordinates, with residual stored in the E₈ codeword. Reconstruction error \(< 2^{-15}\).

// vesper_autopoietic.ts — Mode selector S(E,C), audit chain
export type Mode = 'A'|'B'|'C'|'D';

export function hamiltonian(s: State): number {
  const {t1,w1,t2,w2,m1=1,m2=1,L1=1,L2=1,g=9.81} = s;
  const T = 0.5*m1*L1*L1*w1*w1 + 0.5*m2*(L1*L1*w1*w1 + L2*L2*w2*w2
    + 2*L1*L2*w1*w2*Math.cos(t1-t2));
  const V = -m1*g*L1*Math.cos(t1) - m2*g*(L1*Math.cos(t1)+L2*Math.cos(t2));
  return T+V;
}

export function chaoticity(s: State): number {
  return Math.abs(s.w1-s.w2) + 0.5*Math.abs(Math.sin(s.t1-s.t2));
}

export function select_mode(s: State): Mode {
  const E = hamiltonian(s);
  const C = chaoticity(s);
  if (E > 4.0 && C > 3.0) return 'D';
  if (C >= 2.0) return 'C';
  if (E >= 2.0) return 'B';
  return 'A';
}

export class AuditChain {
  prev = '00'.repeat(32);
  log: any[] = [];
  append(step: number, mode: Mode, stateHash: string) {
    const entry = sha3_256(this.prev + '|' + step + '|' + mode + '|' + stateHash);
    this.log.push({step, mode, hash: entry});
    this.prev = entry;
    return entry;
  }
}
